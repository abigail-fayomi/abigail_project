import pandas as pd

education = pd.read_csv("data/female_education.csv")

education = education.rename(columns={"name": "country"})

education = education.melt(
    id_vars=["geo", "country"],
    var_name="year",
    value_name="education"
)

education["year"] = pd.to_numeric(education["year"], errors="coerce")
education["education"] = pd.to_numeric(education["education"], errors="coerce")

education = education.dropna(subset=["year", "education"])

# optional: make year integer
education["year"] = education["year"].astype(int)
